Bài 1: Viết lại cấu trúc if sau thành cấu trúc switch


let browser = prompt("Enter browser name!") if (browser == 'Edge') { alert("You've got the Edge!"); } else if (browser == 'Chrome' || browser == 'Firefox' || browser == 'Safari' || browser == 'Opera') { alert('Okay we support these browsers too'); } else { alert('We hope that this page looks ok!'); }

Bài 2: Viết lại cấu trúc if sau thành cấu trúc switch

let a = prompt('a?', ''); if (a == 0) { alert(0); } if (a == 1) { alert(1); } if (a == 2 || a == 3) { alert('2,3'); }
